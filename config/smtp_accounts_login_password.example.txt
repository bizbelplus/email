# Формат: одна строка = login:password

# Приложение попробует определить SMTP-хост по домену email автоматически.
# Для нестандартных доменов настройте config/smtp_domains.txt или задайте host/port в основном config/settings.yaml.
#
# Примеры:
example@gmail.com:app_password_here
example@outlook.com:your_password_here
example@mail.ru:your_password_here
